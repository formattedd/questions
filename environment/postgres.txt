  db:
    image: postgres:latest
    restart: always
    ports:

      POSTGRES_PASSWORD: 'password'
      POSTGRES_USER: 'user'
      POSTGRES_DB: 'postgres'
      PGDATA: '/var/lib/postgresql/data'
    volumes:
      - ./postgres:/var/lib/postgresql/data
